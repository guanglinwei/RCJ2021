44 struct Block
45 {
46  // print block structure!
47  void print()
48  {
49  int i, j;
50  char buf[128], sig[6], d;
51  bool flag;
52  if (m_signature>CCC_MAX_SIGNATURE) // color code! (CC)
53  {
54  // convert signature number to an octal string
55  for (i=12, j=0, flag=false; i>=0; i-=3)
56  {
57  d = (m_signature>>i)&0x07;
58  if (d>0 && !flag)
59  flag = true;
60  if (flag)
61  sig[j++] = d + '0';
62  }
63  sig[j] = '\0';
64  sprintf(buf, "CC block sig: %s (%d decimal) x: %d y: %d width: %d height: %d angle: %d index: %d age: %d", sig, m_signature, m_x, m_y, m_width, m_height, m_angle, m_index, m_age);
65  }
66  else // regular block. Note, angle is always zero, so no need to print
67  sprintf(buf, "sig: %d x: %d y: %d width: %d height: %d index: %d age: %d", m_signature, m_x, m_y, m_width, m_height, m_index, m_age);
68  Serial.println(buf);
69  }

100 template <class LinkType> int8_t Pixy2CCC<LinkType>::getBlocks(bool wait, uint8_t sigmap, uint8_t maxBlocks)
101 {
102  blocks = NULL;
103  numBlocks = 0;
104 
105  while(1)
106  {
107  // fill in request data
108  m_pixy->m_bufPayload[0] = sigmap;
109  m_pixy->m_bufPayload[1] = maxBlocks;
110  m_pixy->m_length = 2;
111  m_pixy->m_type = CCC_REQUEST_BLOCKS;
112 
113  // send request
114  m_pixy->sendPacket();
115  if (m_pixy->recvPacket()==0)
116  {
117  if (m_pixy->m_type==CCC_RESPONSE_BLOCKS)
118  {
119  blocks = (Block *)m_pixy->m_buf;
120  numBlocks = m_pixy->m_length/sizeof(Block);
121  return numBlocks;
122  }
123  // deal with busy and program changing states from Pixy (we'll wait)
124  else if (m_pixy->m_type==PIXY_TYPE_RESPONSE_ERROR)
125  {
126  if ((int8_t)m_pixy->m_buf[0]==PIXY_RESULT_BUSY)
127  {
128  if(!wait)
129  return PIXY_RESULT_BUSY; // new data not available yet
130  }
131  else if ((int8_t)m_pixy->m_buf[0]!=PIXY_RESULT_PROG_CHANGING)
132  return m_pixy->m_buf[0];
133  }
134  }
135  else
136  return PIXY_RESULT_ERROR; // some kind of bitstream error
137 
138  // If we're waiting for frame data, don't thrash Pixy with requests.
139  // We can give up half a millisecond of latency (worst case)
140  delayMicroseconds(500);
141  }
142 }
