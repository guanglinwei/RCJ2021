77 /*
78  * Pointer swapping:
79  * sext. r g b r<>b g<>b r <> g result
80  * 0 0 0 v u c !u v c u v c
81  * 0 0 1 d v c d v c
82  * 0 1 0 c v u u v c u v c
83  * 0 1 1 c d v v d c d v c d v c
84  * 1 0 0 u c v u v c u v c
85  * 1 0 1 v c d v d c d v c d v c
86  *
87  * if(sextant & 2)
88  * r <-> b
89  *
90  * if(sextant & 4)
91  * g <-> b
92  *
93  * if(!(sextant & 6) {
94  * if(!(sextant & 1))
95  * r <-> g
96  * } else {
97  * if(sextant & 1)
98  * r <-> g
99  * }
100  */
101 #define HSV_SWAPPTR(a,b) do { uint8_t *tmp = (a); (a) = (b); (b) = tmp; } while(0)
102 #define HSV_POINTER_SWAP(sextant,r,g,b) \
103  do { \
104  if((sextant) & 2) { \
105  HSV_SWAPPTR((r), (b)); \
106  } \
107  if((sextant) & 4) { \
108  HSV_SWAPPTR((g), (b)); \
109  } \
110  if(!((sextant) & 6)) { \
111  if(!((sextant) & 1)) { \
112  HSV_SWAPPTR((r), (g)); \
113  } \
114  } else { \
115  if((sextant) & 1) { \
116  HSV_SWAPPTR((r), (g)); \
117  } \
118  } \
119  } while(0)

125 /* Multiply instruction available */
126 #define MUL(ra,rb,pfx) \
127  "mul " ra ", " rb "\n\t"

129 /*
130  * Small AVR cores (f.x. ATtiny and ATmega*u2) do not have a mul instruction.
131  * It is available from the avr4 architecture.
132  *
133  * Multiply: r1:r0 = ra * rb (clobbers r19, r17, r16)
134  * Algorithm:
135  * uint16_t mul(uint8_t ra, uint8_t rb)
136  * {
137  * r1:r0 = 0
138  * r19 = ra
139  * r17:r16 = rb
140  * do {
141  * if(r19 & 1)
142  * r1:r0 += r17:r16
143  * r17:r16 += r17:r16
144  * r19 >>= 1;
145  * } while(r19)
146  * return r1:r0
147  * }
148  */
149 #define MUL(ra,rb,pfx) \
150  "\n" \
151  ".L" pfx "mul_" ra "_" rb "_%=:\n\t" \
152  "clr r0\n\t" \
153  "clr r1\n\t" \
154  "mov r19, " ra "\n\t" \
155  "mov r16, " rb "\n\t" \
156  "clr r17\n" \
157  ".L" pfx "mul_loop_" ra "_" rb "_%=:\n\t" \
158  "sbrc r19, 0\n\t" \
159  "add r0, r16\n\t" \
160  "sbrc r19, 0\n\t" \
161  "adc r1, r17\n\t" \
162  "add r16, r16\n\t" \
163  "adc r17, r17\n\t" \
164  "lsr r19\n\t" \
165  "brne .L" pfx "mul_loop_" ra "_" rb "_%=\n\t"
166 #endif /* defined(__AVR_HAVE_MUL__) */
