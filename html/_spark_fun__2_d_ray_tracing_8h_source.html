44 // Because the order matters they have an orientation
45 class sf2drt_triplet{
46 protected:
47  public: // temporary
48  sf2drt_coordinate_t _p1;

50  sf2drt_coordinate_t _p3;
51 
52 public:
53  sf2drt_orientation_t orientation( void )
54  {
55  // The cross product way
56  double cross = ((double)_p2.y - (double)_p1.y) * ((double)_p3.x - (double)_p2.x) - ((double)_p2.x - (double)_p1.x) * ((double)_p3.y - (double)_p2.y);
57 
58  if(cross == 0){ return sf2drt_orientation_colinear; }
59  if(cross > 0){ return sf2drt_orientation_negative; }
60  else{ return sf2drt_orientation_positive; }
61  }
62  bool p3contained( void )
63  {
64  if((_p3.x <= max(_p1.x, _p2.x)) && (_p3.x >= min(_p1.x, _p2.x)) && (_p3.y <= max(_p1.y, _p2.y)) && (_p3.y >= min(_p1.y, _p2.y))){ return true; }
65  return false;
66  }

160  bool contains(sf2drt_coordinate_t p)
161  {
162  // Find maximum extent of the polygon in x direction
163  sf2drt_bounds_t bounds = getBounds();
164 
165  if((p.x < bounds.minx) || (p.x > bounds.maxx) || (p.y < bounds.miny) || (p.y > bounds.maxy)){ return false; }
166 
167  // Make a line from the point of interest to the right until the maximum size
168  sf2drt_coordinate_t p2 = p;
169  p2.x = max(bounds.maxx, p2.x);
170  sf2drt_line bisector(p, p2);
171 
172  uint8_t inside = 0x00;
173  sf2drt_max_qty_t indi = 0;
174  for(indi = 0; indi < _num_sides - 1; indi++)
175  {
176  sf2drt_line segment((*(_p_points + indi)), (*(_p_points + indi + 1)));
177  if(bisector.intersects(segment)){ inside ^= 0x01; }
178  }
179  sf2drt_line segment((*(_p_points + indi)), (*(_p_points)));
180  if(bisector.intersects(segment)){ inside ^= 0x01; }
181 
182  if(inside & 0x01)
183  {
184  return true;
185  }
186  return false;
187  }

224  // Return the "center of mass" of the polygon (average location of all vertices, actually)
225  sf2drt_coordinate_t retval = {0,0};

240  retval.y = (sf2drt_extent_t)round(avgY);
